Sayı tahmin (JavaScript)
let secret = Math.floor(Math.random() * 100) + 1;
const input = document.getElementById("guess-input");
const btn = document.getElementById("guess-btn");
const out = document.getElementById("guess-output");

btn.addEventListener("click", () => {
  const val = Number(input.value);
  if (!val || val < 1 || val > 100) {
    out.textContent = "1–100 arasında bir sayı yaz.";
    out.style.color = "#f97316";
    return;
  }
  if (val === secret) {
    out.textContent = "Doğru! Yeni sayı tuttum.";
    out.style.color = "#22c55e";
    secret = Math.floor(Math.random() * 100) + 1;
  } else if (val < secret) {
    out.textContent = "Daha büyük dene.";
    out.style.color = "#eab308";
  } else {
    out.textContent = "Daha küçük dene.";
    out.style.color = "#eab308";
  }
});
