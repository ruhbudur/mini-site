JavaScript – Butona tıklayınca arka plan rengi değiştir
const colors = ["#0f172a", "#1d4ed8", "#16a34a", "#be123c"];

document.querySelector("button").addEventListener("click", () => {
  const secilen = colors[Math.floor(Math.random() * colors.length)];
  document.body.style.backgroundColor = secilen;
});
C++ – Dizideki en büyük eleman
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> a = {3, 7, 2, 9, 4};
    int maks = a[0];
    for (int x : a) {
        if (x > maks) maks = x;
    }
    cout << "En büyük:" << maks << endl;
    return 0;
}
