5. Debounce (yazmayı bitirince tetikle)
function debounce(fn, ms) {
  let t;
  return (...args) => {
    clearTimeout(t);
    t = setTimeout(() => fn(...args), ms);
  };
}
const aramaYap = debounce(() => console.log("Aranıyor..."), 300);
input.addEventListener("input", aramaYap);

Arama kutusu gibi yerlerde her tuşta değil, yazı durunca işlem yapmak için.

C++

1. Dizideki en büyük eleman
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> a = {3, 7, 2, 9, 4};
    int maks = a[0];
    for (int x : a) {
        if (x > maks) maks = x;
    }
    cout << "En büyük:" << maks << endl;
    return 0;
}

Dizilerle çalışırken en temel arama örneklerinden biri.

2. Faktöriyel (recursive)
#include <iostream>
using namespace std;

int faktoriyel(int n) {
    if (n <= 1) return 1;
    return n * faktoriyel(n - 1);
}
int main() {
    cout << "5! = " << faktoriyel(5) << endl;
    return 0;
}

Recursion ve matematiksel fonksiyon örneği.

4. İkili arama (binary search)
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    vector<int> v = {2, 5, 8, 12, 16};
    int aranan = 8;
    if (binary_search(v.begin(), v.end(), aranan))
        cout << "Bulundu" << endl;
    else
        cout << "Yok" << endl;
    return 0;
}
